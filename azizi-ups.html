<script>

const deviceValues = {

'desktop': { current: 1.5, power: 330 },

'fridge': { current: 2.5, power: 550 },

'server': { current: 4, power: 880 },

'tv': { current: 0.7, power: 150 },

'lighting': { current: 0.2, power: 44 }

};

 

function fillDeviceValues() {

const device = document.getElementById('device-select').value;

if (device !== 'custom') {

document.getElementById('current').value = deviceValues[device].current;

}

}

 

function calculate() {

// دریافت مقادیر ورودی

const current = parseFloat(document.getElementById('current').value);

const voltage = parseFloat(document.getElementById('voltage').value);

const hours = parseFloat(document.getElementById('hours').value);

const batteryVoltage = parseFloat(document.getElementById('battery-voltage').value);

 

// محاسبات

const powerFactor = 0.8;

const power = current * voltage * powerFactor;

const requiredPower = power * hours;

const batteryCapacity = requiredPower / batteryVoltage;

const recommendedUPS = power * 1.25; // 25% اضافه

 

// نمایش نتایج

document.getElementById('result').innerHTML = `

<h3>نتایج محاسبه:</h3>

<p>توان مصرفی: <strong>${power.toFixed(2)} وات</strong></p>

<p>انرژی مورد نیاز: <strong>${requiredPower.toFixed(2)} وات-ساعت</strong></p>

<p>ظرفیت باتری مورد نیاز: <strong>${batteryCapacity.toFixed(2)} آمپر-ساعت (در ${batteryVoltage}V)</strong></p>

<p>پیشنهاد یو پی اس: <strong>${recommendedUPS.toFixed(2)} وات</strong> (با احتساب 25% اضافه)</p>

`;

}

</script>
